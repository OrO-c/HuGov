{{- /* image.html — 图片处理 partial resources.Get 从 assets/ 找图片，找不到时 fallback 输出原始 src 找到时: WebP + 3种srcset尺寸 + picture降级 + width/height 找不到时: 输出原始 img + loading/fetchpriority + 靠CSS aspect-ratio 防CLS 参数: src — 图片路径 (如 "/images/gate.jpg" 或 "images/gate.jpg") alt — 替代文本 class — CSS类名 lazy — 是否懒加载 (bool, 默认true) fetchpriority — 优先级 ("high" / "auto", 默认"auto") link — 外层包裹链接 (可选) */ -}} {{- $src := .src }} {{- $alt := .alt | default "图片" }} {{- $class := .class | default "" }} {{- $lazy := .lazy | default true }} {{- $fp := .fetchpriority | default "auto" }} {{- $link := .link | default "" }} {{- /* 从 assets/ 找图 */}} {{- $cleanSrc := $src }} {{- if hasPrefix $cleanSrc "/" }} {{- $cleanSrc = strings.TrimPrefix "/" $cleanSrc }} {{- end }} {{- $img := resources.Get $cleanSrc }} {{- if $img }} {{- /* 资源存在 → 全量优化 */}} {{- $w := $img.Width }} {{- $h := $img.Height }} {{- $srcsetItems := slice }} {{- $sizesList := slice "400" "800" "1200" }} {{- range $sizesList }} {{- $sw := . | int }} {{- if ge $w $sw }} {{- $webpResized := $img.Resize (printf "%dx webp" $sw) }} {{- $srcsetItems = $srcsetItems | append (printf "%s %sw" $webpResized.RelPermalink $sw) }} {{- end }} {{- end }} {{- $fullWebp := $img.Resize (printf "%dx%d webp" $w $h) }} {{- $fullAvif := $img.Resize (printf "%dx%d avif" $w $h) }} {{- if $link }}{{ end -}} {{- if $srcsetItems }} {{- else }} {{- end }} {{- if $link }}{{ end -}} {{- else }} {{- /* 资源不存在 → fallback */}} {{- if $link }}{{ end -}} {{- if $link }}{{ end -}} {{- end -}}
